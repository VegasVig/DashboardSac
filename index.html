O script precisa aceitar GET com ?action=getDados e retornar JSON com os registros.

Estrutura Esperada do Google Sheets

A planilha deve ter as colunas na seguinte ordem:
protocolo | nome | email | telefone | cpf | nota | motivos | descricao | decisao | data_hora

O Apps Script precisa publicar como Web App com acesso Anyone e retornar JSON assim: 

const SHEET_NAME = 'SAC';

function doGet(e) {
  const cb = e.parameter.callback || '';
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
                  .getSheetByName(SHEET_NAME) ||
                SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

  const rows    = sheet.getDataRange().getValues();
  const headers = rows[0];  // Data/Hora, Protocolo, Nome, CPF/CNPJ, Telefone...
  const data    = rows.slice(1).map(r => {
    const obj = {};
    headers.forEach((h, i) => { obj[String(h)] = r[i]; });
    return obj;
  });

  const json = JSON.stringify({ data });
  const output = cb ? cb + '(' + json + ')' : json;
  const mime   = cb ? ContentService.MimeType.JAVASCRIPT
                    : ContentService.MimeType.JSON;

  return ContentService.createTextOutput(output).setMimeType(mime);
}
